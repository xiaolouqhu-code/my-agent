# 🎯 流金亚马逊实时数据监控系统 - 架构设计文档

**项目代号**: `flowgold-amz-monitor-v1`
**设计时间**: 2025-10-26
**设计目标**: 解决数据滞后、决策缓慢、止损不及时的核心问题

---

## 📊 系统概览

### 核心目标
1. **实时监控**：8个店铺、919个ASIN的关键指标每日自动更新
2. **智能预警**：自动识别断货、ACOS异常、高退款率等风险
3. **决策支持**：基于数据自动生成运营建议
4. **降本增效**：减少手动数据收集，提高决策速度

### 系统定位
```
替代：手动截图PDF → 自动化数据报告
升级：静态分析 → 实时监控 + 智能预警
赋能：被动决策 → 主动建议 + 快速响应
```

## 📊 核心KPI计算逻辑

### 1. 毛利率计算
```python
def calculate_gross_margin(revenue: float, cost: float, fees: float) -> float:
    """
    计算毛利率
    毛利率 = (销售额 - 成本 - 费用) / 销售额 * 100%
    """
    profit = revenue - cost - fees
    margin = (profit / revenue * 100) if revenue > 0 else 0
    return round(margin, 2)
```

### 2. ACOS计算
```python
def calculate_acos(ad_spend: float, ad_sales: float) -> float:
    """
    ACOS = 广告花费 / 广告销售额 * 100%
    """
    acos = (ad_spend / ad_sales * 100) if ad_sales > 0 else 100
    return round(acos, 2)
```

### 3. ROAS计算
```python
def calculate_roas(ad_sales: float, ad_spend: float) -> float:
    """
    ROAS = 广告销售额 / 广告花费
    """
    roas = ad_sales / ad_spend if ad_spend > 0 else 0
    return round(roas, 2)
```

### 4. 库存天数计算
```python
def calculate_days_of_inventory(
    available_qty: int,
    avg_daily_sales: float
) -> int:
    """
    可售天数 = FBA可售数量 / 日均销量
    """
    days = int(available_qty / avg_daily_sales) if avg_daily_sales > 0 else 999
    return days
```

### 5. 产品分级算法
```python
def grade_product(
    gross_margin: float,
    roas: float,
    sales_rank: int,
    return_rate: float
) -> str:
    """
    产品分级逻辑
    S级: 毛利率>25% & ROAS>4 & 退款率<3%
    A级: 毛利率15-25% & ROAS>3
    B级: 毛利率5-15% 或 ROAS 2-3
    C级: 毛利率<5% 或 ROAS<2 或 退款率>8%
    """
    if gross_margin > 25 and roas > 4 and return_rate < 3:
        return "S"
    elif gross_margin >= 15 and roas >= 3:
        return "A"
    elif gross_margin >= 5 or roas >= 2:
        return "B"
    else:
        return "C"
```

### 预警检查流程

```python
# alerts/checker.py

async def check_alerts():
    """每小时执行一次预警检查"""

    # 1. 获取所有产品的最新数据
    products = await db.get_all_products_with_metrics()

    # 2. 应用预警规则
    alerts = []
    for product in products:
        for level, rules in ALERT_RULES.items():
            for rule in rules:
                if rule["condition"](product):
                    alert = {
                        "level": level,
                        "type": rule["name"],
                        "asin": product["asin"],
                        "message": rule["message"].format(**product),
                        "action": rule["action"],
                        "priority": rule["priority"],
                        "created_at": datetime.now()
                    }
                    alerts.append(alert)

    # 3. 保存预警到数据库
    await db.save_alerts(alerts)

    # 4. 发送通知（可选）
    await send_notifications(alerts)

    return alerts
```

## 🗄️ 数据库Schema设计

### PostgreSQL表结构

```sql
-- 店铺表
CREATE TABLE stores (
    id SERIAL PRIMARY KEY,
    store_id VARCHAR(50) UNIQUE NOT NULL,
    store_name VARCHAR(100) NOT NULL,
    marketplace VARCHAR(10) NOT NULL, -- US, CA, UK, DE, AE
    company_type VARCHAR(50), -- 自营、合作、合资
    created_at TIMESTAMP DEFAULT NOW()
);

-- 产品表
CREATE TABLE products (
    id SERIAL PRIMARY KEY,
    asin VARCHAR(20) NOT NULL,
    sku VARCHAR(100),
    store_id VARCHAR(50) REFERENCES stores(store_id),
    product_name VARCHAR(500),
    brand VARCHAR(100),
    category VARCHAR(200),
    cost DECIMAL(10,2), -- 采购成本
    current_price DECIMAL(10,2),
    grade CHAR(1), -- S, A, B, C
    status VARCHAR(20), -- active, paused, discontinued
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(asin, store_id)
);

-- 每日指标表
CREATE TABLE daily_metrics (
    id SERIAL PRIMARY KEY,
    date DATE NOT NULL,
    asin VARCHAR(20) NOT NULL,
    store_id VARCHAR(50) NOT NULL,

    -- 销售指标
    sales_qty INT DEFAULT 0,
    revenue DECIMAL(10,2) DEFAULT 0,
    orders INT DEFAULT 0,

    -- 广告指标
    ad_spend DECIMAL(10,2) DEFAULT 0,
    ad_sales DECIMAL(10,2) DEFAULT 0,
    clicks INT DEFAULT 0,
    impressions INT DEFAULT 0,

    -- 利润指标
    cost DECIMAL(10,2) DEFAULT 0,
    fees DECIMAL(10,2) DEFAULT 0,
    profit DECIMAL(10,2) DEFAULT 0,
    gross_margin DECIMAL(5,2), -- 毛利率%

    -- 库存指标
    fba_available INT DEFAULT 0,
    days_of_inventory INT,

    -- 客户指标
    return_qty INT DEFAULT 0,
    return_rate DECIMAL(5,2),
    review_count INT DEFAULT 0,
    review_rating DECIMAL(3,2),

    -- 计算指标
    acos DECIMAL(5,2),
    roas DECIMAL(5,2),

    created_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(date, asin, store_id)
);

-- 预警表
CREATE TABLE alerts (
    id SERIAL PRIMARY KEY,
    level VARCHAR(20) NOT NULL, -- CRITICAL, WARNING, OPPORTUNITY
    type VARCHAR(50) NOT NULL, -- stock_out, high_acos, etc.
    asin VARCHAR(20) NOT NULL,
    store_id VARCHAR(50) NOT NULL,
    message TEXT NOT NULL,
    action TEXT,
    priority INT,
    status VARCHAR(20) DEFAULT 'active', -- active, resolved, ignored
    created_at TIMESTAMP DEFAULT NOW(),
    resolved_at TIMESTAMP
);

-- 产品分级历史表
CREATE TABLE product_grades_history (
    id SERIAL PRIMARY KEY,
    asin VARCHAR(20) NOT NULL,
    date DATE NOT NULL,
    grade CHAR(1),
    gross_margin DECIMAL(5,2),
    roas DECIMAL(5,2),
    reason TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);

-- 索引
CREATE INDEX idx_daily_metrics_date ON daily_metrics(date);
CREATE INDEX idx_daily_metrics_asin ON daily_metrics(asin);
CREATE INDEX idx_alerts_level ON alerts(level);
CREATE INDEX idx_alerts_status ON alerts(status);
CREATE INDEX idx_products_grade ON products(grade);
```
